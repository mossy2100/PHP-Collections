19/**
20 * A type-specific list implementation.
21 *
22 * 1. Indexes are always sequential integers starting from 0, as with PHP lists. The largest index (a.k.a. offset or
23 * key) will equal the number of items in the Sequence minus 1.
24 *
25 * 2. Allowed types for Sequence values can be specified in the constructor, enabling the Sequence to function like a
26 * generic array (or equivalent) in C# or Java, i.e. `new Sequence('int')` is equivalent to `new List<int>` in C#.
27 * Allowed types must be specified as strings (with union types and nullable syntax supported), or an iterable with
28 * values equal to type names. See TypeSet for more details.
29 * (NB: Intersection types and DNF syntax are currently NOT supported.)
30 *
31 * 3. Sequence items can be set at positions beyond the current range, but intermediate items will be filled in with a
32 * default value. Sensible defaults are used for common types, or a default value can be specified in the constructor.
33 * If the default value is an object, it will be cloned as needed.
34 *
35 * 4. Array-like access using square brackets [] and iteration using foreach is supported via the ArrayAccess and
36 * Iterator interfaces.
37
38 * ## Automatic Defaults
39 * For value types, the following defaults are used:
40 * - null or mixed → null
41 * - int, uint, number, or scalar → 0
42 * - float → 0.0
43 * - string → '' (empty string)
44 * - bool → false
45 * - array → [] (empty array)
46 * If the allowed types only permit specific classes, interfaces, or traits, then a default value of a matching type
47 * must be provided, which will be cloned as needed.
48 *
49 * @example Basic usage
50 * $strings = new Sequence('string');
51 * $strings->append('hello');
52 * $strings[] = 'world';
53 *
54 * @example Union types with custom default
55 * $mixed = new Sequence('string|int', 'default');
56 * $mixed->append('text');
57 * $mixed->append(42);
58 *
59 * @example Object types
60 * $dates = new Sequence('DateTime', new DateTime());
61 * $dates->append(new DateTime('tomorrow'));
62 *
63 * @example Interface types
64 * $countables = new Sequence('Countable', []);
65 * $countables->append([1, 2, 3]);         // Arrays are countable
66 * $countables->append(new ArrayObject()); // ArrayObject implements Countable
67 */

157    /**
158     * Generate a new Sequence of numbers spanning a given range, using a given step size.
159     *
160     * A step size of 0 is not allowed, as this would cause an infinite loop.
161     *
162     * This method is analogous to range(), except it does not support strings.
163     * @see https://www.php.net/manual/en/function.range.php
164     *
165     * Note: When using floats, be aware of floating-point precision issues.
166     * The final value may not be exactly $end due to rounding errors.
167     *
168     * @example
169     * Sequence::range(1, 10);          // [1, 2, 3, ..., 10]
170     * Sequence::range(0, 1, 0.1);      // [0.0, 0.1, 0.2, ..., 1.0]
171     * Sequence::range(10, 1, -1);      // [10, 9, 8, ..., 1]
172     *
173     * @param int|float $start The start of the range.
174     * @param int|float $end The end of the range.
175     * @param int|float $step The step size (default 1).
176     * @return self The new Sequence.
177     * @throws ValueError If the step size is invalid for the range specified.
178     */
179    public static function range(int|float $start, int|float $end, int|float $step = 1): self
180    {
181        // Validate step size. Use loose comparison here to validate either an int or float argument.
182        if ($step == 0) {
183            throw new ValueError("The step size cannot be zero.");
184        }
185        if ($start <= $end && $step < 0) {
186            throw new ValueError("The step size must be positive for an increasing range.");
187        }
188        if ($start >= $end && $step > 0) {
189            throw new ValueError("The step size must be negative for a decreasing range.");
190        }
191
192        // If any of the arguments are floats, generate a Sequence of floats; otherwise, ints.
193        $type = is_float($start) || is_float($end) || is_float($step) ? 'float' : 'int';
194
195        // Construct the new Sequence.
196        $seq = new self($type);
197
198        // Cast $start to float, if necessary, to ensure all generated values are also float.
199        if ($type === 'float') {
200            $start = (float)$start;
201        }
202
203        // Generate the range.
204        if ($step > 0) {
205            // Ascending.
206            for ($i = $start; $i <= $end; $i += $step) {
207                $seq->append($i);
208            }
209        } else {
210            // Descending.
211            for ($i = $start; $i >= $end; $i += $step) {
212                $seq->append($i);
213            }
214        }
215
216        // Return the new Sequence.
217        return $seq;
218    }

331        // Check the item type.
332        $this->valueTypes->check($item);
333
334        // Check the index is valid.
335        $this->checkIndex($index, false);
336
337        // For indexes beyond the end of the Sequence, no items need to be shifted, so defer to offsetSet().
338        // That will take care of gap-filling with default values.
339        $orig_count = count($this->items);
340        if ($index >= $orig_count) {

342            return $this;
343        }

345        // Shift elements after $index right by 1.
346        for ($j = $orig_count; $j > $index; $j--) {
347            $this->items[$j] = $this->items[$j - 1];
348        }
349
350        // Set the new value of the item at $index.
351        $this->items[$index] = $item;
352
353        // Return this for chaining.
354        return $this;
355    }
