18/**
19 * A type-specific list implementation.
20 *
21 * 1. Indexes are always sequential integers starting from 0, as with PHP lists. The largest index (a.k.a. offset or
22 * key) will equal the number of items in the Sequence minus 1.
23 *
24 * 2. Allowed types for Sequence values can be specified in the constructor, enabling the Sequence to function like a
25 * generic array (or equivalent) in C# or Java, i.e. `new Sequence('int')` is equivalent to `new List<int>` in C#.
26 * Allowed types can be specified as strings (with union types and nullable syntax supported), or an iterable of string
27 * values equal to type names. These are two different ways to specify multiple types and can't be combined; so you
28 * can't specify, for example, `['int|string', '?float]`. Specifying null for the allowed types (not 'null') means
29 * values of any type are allowed. See TypeSet for more details.
30 * (NB: Intersection types and DNF syntax are currently NOT supported.)
31 *
32 * 3. Sequence items can be set at positions beyond the current range, but intermediate items will be filled in with a
33 * default value. Sensible defaults are used for common types, or a default value can be specified in the constructor.
34 * If the default value is an object, it will be cloned as needed.
35 *
36 * 4. Array-like access using square brackets [] and iteration using foreach is supported via the ArrayAccess and
37 * Iterator interfaces.
38 *
39 * ## Automatic Defaults
40 * For value types, the following defaults are used:
41 * - null or mixed  null
42 * - int, uint, number, or scalar  0
43 * - float  0.0
44 * - string  '' (empty string)
45 * - bool  false
46 * - array or iterable  [] (empty array)
47 * For other types (i.e. resource, object, interface, callable), the automatic default value will be null, and if 'null'
48 * has not been specified in the allowed types, it will be automatically added.
49 *
50 * ## Implementation Notes
51 * Several methods use `func_num_args()` to distinguish between "parameter omitted" and "parameter explicitly set to
52 * null". This is necessary because PHP doesn't provide a native way to make this distinction with optional parameters.
53 *
54 * @example Basic usage
55 * $strings = new Sequence('string');
56 * $strings->append('hello');
57 * $strings[] = 'world';
58 *
59 * @example Union types with custom default
60 * $mixed = new Sequence('string|int', 'default');
61 * $mixed->append('text');
62 * $mixed->append(42);
63 *
64 * @example Object types
65 * $dates = new Sequence('DateTime', new DateTime());
66 * $dates->append(new DateTime('tomorrow'));
67 *
68 * @example Interface types
69 * $countables = new Sequence('Countable', []);
70 * $countables->append([1, 2, 3]);         // Arrays are countable
71 * $countables->append(new ArrayObject()); // ArrayObject implements Countable
72 */

92     * - null = Values of any type are allowed.
93     * - string = A type name, or multiple types using union type or nullable type syntax, e.g. 'string', 'int|null', '?int'

98     * effect of calling insert() or offsetSet() (either directly or via square bracket syntax).

101     * It doesn't really make sense to have a resource as a default value, but it's allowed.
102     *

106     * @param mixed $default_value Default value for new items (default null).

109     * @throws TypeError If a type is not specified as a string, or any imported values have disallowed types.

159     * Generate a new Sequence of numbers spanning a given range, using a given step size.
160     *
161     * A step size of 0 is not allowed, as this would cause an infinite loop.
162     *
163     * This method is analogous to range(), except it does not support strings.

165     *
166     * Note: When using floats, be aware of floating-point precision issues.
167     * The final value may not be exactly $end due to rounding errors.
168     *
169     * @example
170     * Sequence::range(1, 10);          // [1, 2, 3, ..., 10]
171     * Sequence::range(0, 1, 0.1);      // [0.0, 0.1, 0.2, ..., 1.0]
172     * Sequence::range(10, 1, -1);      // [10, 9, 8, ..., 1]
173     *
174     * @param int|float $start The start of the range.
175     * @param int|float $end The end of the range.
176     * @param int|float $step The step size (default 1).
177     * @return self The new Sequence.
178     * @throws ValueError If the step size is invalid for the range specified.
179     */
180    public static function range(int|float $start, int|float $end, int|float $step = 1): self
181    {
182        // Validate step size. Use loose comparison here to validate either an int or float argument.
183        if ($step == 0) {
184            throw new ValueError("The step size cannot be zero.");
185        }
186        if ($start <= $end && $step < 0) {
187            throw new ValueError("The step size must be positive for an increasing range.");
188        }
189        if ($start >= $end && $step > 0) {
190            throw new ValueError("The step size must be negative for a decreasing range.");
191        }
192
193        // If any of the arguments are floats, generate a Sequence of floats; otherwise, ints.
194        $type = is_float($start) || is_float($end) || is_float($step) ? 'float' : 'int';
195
196        // Construct the new Sequence.
197        $seq = new self($type);
198
199        // Cast $start to float, if necessary, to ensure all generated values are also float.
200        if ($type === 'float') {
201            $start = (float)$start;
202        }
203
204        // Generate the range.
205        if ($step > 0) {
206            // Ascending.
207            for ($i = $start; $i <= $end; $i += $step) {
208                $seq->append($i);
209            }
210        }
211        else {
212            // Descending.
213            for ($i = $start; $i >= $end; $i += $step) {
214                $seq->append($i);
215            }
216        }
217
218        // Return the new Sequence.
219        return $seq;
220    }

292     * $sequence->append($item);
293     * $sequence->append($item1, $item2, $item3);
294     * $sequence->append(...$items);
295     */
296    public function append(mixed ...$items): self
297    {
298        foreach ($items as $item) {
299            // Check the item type.
300            $this->valueTypes->check($item);
301
302            // Append an element to the end of the Sequence.
303            $this->items[] = $item;
304        }
305
306        // Return this for chaining.
307        return $this;
308    }

310    /**
311     * Add one or more items to the start of the Sequence.
312     *
313     * NB: This is a mutating method.
314     *
315     * Note that the list of elements is prepended as a whole, so that the prepended elements stay in the same order.
316     * This behaviour parallels array_unshift().
317     * @see https://www.php.net/manual/en/function.array-unshift.php
318     *
319     * @param mixed ...$items The items to add to the Sequence.
320     * @return $this The Sequence instance.
321     *
322     * @example
323     * $sequence->prepend($item);
324     * $sequence->prepend($item1, $item2, $item3);
325     * $sequence->prepend(...$items);
326     */
327    public function prepend(mixed ...$items): self
328    {
329        // Go through the items in reverse order.
330        for ($i = count($items) - 1; $i >= 0; $i--) {
331            $item = $items[$i];
332
333            // Check the item type.
334            $this->valueTypes->check($item);
335
336            // Prepend an element at the start of the Sequence.
337            array_unshift($this->items, $item);
338        }
339
340        // Return this for chaining.
341        return $this;
342    }

344    /**
345     * Insert an item at the specified position. Later items will be shifted right.
346     *
347     * NB: This is a mutating method.
348     *
349     * @param int $index The zero-based index position to insert the item at.
350     * @param mixed $item The item to insert.
351     * @return $this The Sequence instance.
352     * @throws OutOfRangeException If the index is less than 0.
353     */
354    public function insert(int $index, mixed $item): self
355    {
356        // Check the item type.
357        $this->valueTypes->check($item);
358
359        // Check the index is valid.
360        $this->checkIndex($index, false);
361
362        // For indexes beyond the end of the Sequence, no items need to be shifted, so defer to offsetSet().
363        // That will take care of gap-filling with default values.
364        $orig_count = count($this->items);
365        if ($index >= $orig_count) {
366            $this->offsetSet($index, $item);
367            return $this;
368        }
369
370        // Shift elements after $index right by 1.
371        for ($j = $orig_count; $j > $index; $j--) {
372            $this->items[$j] = $this->items[$j - 1];
373        }
374
375        // Set the new value of the item at $index.
376        $this->items[$index] = $item;
377
378        // Return this for chaining.
379        return $this;
380    }
