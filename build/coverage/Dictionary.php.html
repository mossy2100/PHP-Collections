426     * The resulting Dictionary will have the same type constraints, and will only contain the key-value pairs that
427     * the filter callback returns true for.
428     *
429     * The callback must accept two parameters, for the key and the value, and return a bool.
430     * It can accept more than two parameters, but any additional parameters must be optional.
431     * Also, the callback's parameter types should match the dictionary's allowed key and value types.
432     *
433     * @param callable $callback A callback function that accepts a key and a value, and returns a bool.
434     * @return self A new dictionary with the kept key-value pairs.
435     * @throws TypeError If the callback's parameter types don't match the dictionary's key and value types.
436     * Note also that the callback could throw other kinds of exceptions, or they could throw a TypeError for some
437     * other reason.
438     */
439    #[Override]
440    public function filter(callable $callback): static
441    {
442        // Create a new dictionary with the same type constraints.
443        $result = new self($this->keyTypes, $this->valueTypes);
444
445        // Apply the filter with validation.
446        foreach ($this->items as $item) {
447            // See if we want to keep this pair.
448            $keep = $callback($item->key, $item->value);
449
450            // Validate the result of the callback.
451            if (!is_bool($keep)) {
452                throw new TypeError("The filter callback must return a bool, got " . Types::getBasicType($keep) . ".");
453            }
454
455            // Add pair to keep to the result dictionary.
456            if ($keep) {
457                $result[$item->key] = $item->value;
458            }
459        }
460
461        return $result;
462    }
