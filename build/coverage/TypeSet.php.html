17/**
18 * Encapsulates a set of types, represented as strings.
19 *
20 * ## Supported Type Specifications
21 *
22 * ### Basic Types
23 * - 'null' - Null values
24 * - 'bool' - Boolean values
25 * - 'int' - Integer values
26 * - 'float' - Floating point values
27 * - 'string' - String values
28 * - 'array' - Array values
29 * - 'object' - Any object instance
30 * - 'resource' - Any resource type
31 *
32 * ### Pseudotypes
33 * - 'mixed' - Any type (no restrictions)
34 * - 'scalar' - Any scalar type (string|int|float|bool)
35 * - 'iterable' - Arrays, iterators, generators (anything iterable)
36 * - 'callable' - Functions, methods, closures, invokables
37 *
38 * ### Custom pseudotypes (invented for this package)
39 * - 'number' - Either number type (int|float)
40 * - 'uint' - Unsigned integer type (int where value >= 0)
41 *
42 * ### Resource types
43 * Examples: 'resource (stream)', 'resource (curl)', etc.
44 * These must be given as the "resource (xxx)" format, as returned from get_debug_type(), to distinguish them from class names.
45 *
46 * ### Class, interface and trait names
47 * - Class names: 'DateTime', 'MyNameSpace\MyClass' (leading '\' optional; includes inheritance)
48 * - Interface names: 'Countable', 'JsonSerializable' (includes implementations)
49 * - Leading backslashes can be included, but will be stripped; e.g. \DateTime is stored and matched as DateTime.
50 * - Values matched against a class name will also match parent classes, interfaces, and traits.
51 *
52 * ### Union Types
53 * Use pipe syntax to allow multiple types:
54 * - 'string|int' - String OR integer values
55 * - 'array|object' - Array OR object values
56 * - 'DateTime|null' - DateTime objects OR null
57 *
58 * ### Nullable Types
59 * Use question mark syntax for nullable types:
60 * - '?string' - Equivalent to 'string|null'
61 * - '?DateTime' - Equivalent to 'DateTime|null'
62 *
63 * ### Unsupported pseudo-types: void, never, false, true, self, static.
64 */

135        if ($this->contains('number') && Types::isNumber($value)) {

176                if (trait_exists($type) && Types::usesTrait($value, $type)) {

178                }
179            }
180        }

359     *
360     * @param string|iterable $types The types to add to the TypeSet.
361     * @return $this The modified TypeSet.
362     * @throws TypeError If a type is not provided as a string.
363     * @throws ValueError If a type name is invalid.
364     */
365    public function add(string|iterable $types): self
366    {
367        // Convert a type string, including union type syntax (e.g. 'string|int'), into an array of type names.
368        if (is_string($types)) {
369            $types = explode('|', $types);
370        }
371
372        // Add types to the set.
373        foreach ($types as $type) {
374            // Check the type.
375            if (!is_string($type)) {
376                throw new TypeError("Types must be provided as strings.");
377            }
378
379            // Trim in case the user did something like 'string | int'.
380            // Even thought _add() will trim the type, we need to do this before checking if the question mark notation
381            // has been used.
382            $type = trim($type);

385            if (strlen($type) > 1 && $type[0] === '?') {
386                // Add null and the type being made nullable.
387                $this->_add('null');
388                $this->_add(substr($type, 1));
389            }
390            else {
391                // Add the type.
392                $this->_add($type);
393            }
394        }
395
396        // Return $this for chaining.
397        return $this;
